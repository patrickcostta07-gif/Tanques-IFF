function registerReading(gas) { const data = { gas: gas.toUpperCase(), max: document.getElementById(`in-${gas}-max`).value, nivel: document.getElementById(`in-${gas}-vol`).value, pressao: document.getElementById(`in-${gas}-pres`).value, responsavel: document.getElementById(`in-${gas}-user`).value || "Não informado", observacao: document.getElementById(`in-${gas}-obs`).value, data: new Date().toLocaleString('pt-BR') }; // Validação simples para não registrar campos em branco if (!data.nivel || !data.pressao) { alert("Por favor, preencha o nível atual e a pressão!"); return; } // Convertendo os dados para o formato que o Google Apps Script aceita nativamente sem problemas de CORS const formData = new URLSearchParams(); for (const key in data) { formData.append(key, data[key]); } fetch(SCRIPT_URL, { method: 'POST', mode: 'no-cors', // Evita o bloqueio de CORS do navegador body: formData, headers: { 'Content-Type': 'application/x-www-form-urlencoded' } }) .then(() => { showToast(`✅ Leitura de ${gas.toUpperCase()} enviada!`); // Limpa os campos após o envio document.getElementById(`in-${gas}-vol`).value = ''; document.getElementById(`in-${gas}-pres`).value = ''; document.getElementById(`in-${gas}-obs`).value = ''; // Aguarda um pequeno momento para a planilha atualizar e recarrega os dados na tela setTimeout(loadData, 1500); }) .catch(err => alert("Erro ao salvar: " + err)); }
